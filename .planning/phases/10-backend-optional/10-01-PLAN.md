public async Task LoadCategoriesAsync()
{
    try
    {
        IsLoading = true;
        var categories = await _categoryService.GetAllCategoriesAsync();
        Categories.Clear();
        foreach (var category in categories)
        {
            Categories.Add(category);
        }
    }
    catch (Exception ex)

        System.Diagnostics.Debug.WriteLine($"Error loading categories: {ex.Message}");

**Step 3: Update LoadEventsAsync with IsLoading and error handling**
Replace the current LoadEventsAsync method (lines 69-78):

```csharp
public async Task LoadEventsAsync()
{
    try
    {
        IsLoading = true;
        var events = await _eventService.GetAllEventsAsync();
        _allEvents = events;
        FilteredEvents.Clear();
        foreach (var evt in events)
        {
            FilteredEvents.Add(evt);
        }
    }
    catch (Exception ex)
    {
        System.Diagnostics.Debug.WriteLine($"Error loading events: {ex.Message}");
    }
    finally
    {
        IsLoading = false;
    }
}
```

Key changes:
- Set IsLoading = true at start of async operations
- Wrap async calls in try-catch for error handling
- Set IsLoading = false in finally block (always executes)
- Use Debug.WriteLine for logging (matches existing patterns)

protected override async void OnAppearing()
{
    base.OnAppearing();
    
    // Add layout animation for smooth transitions
    if (BindingContext is CategoryViewModel vm)
    {
        // Ensure data is loaded
        if (vm.Categories.Count == 0)
        {
            await vm.LoadCategoriesAsync();
            await vm.LoadEventsAsync();
        }
        
        // Set up fade animation for CollectionView items
        // This creates a subtle fade-in effect when items appear
        var fadeAnimation = new Microsoft.Maui.Animations.FadeAnimation
        {
            Duration = 300
        };
        
        // Animation will apply when FilteredEvents changes
        vm.FilteredEvents.CollectionChanged += (s, e) =>
        {
            // Trigger subtle fade effect on collection changes
            // This is a simple approach - more advanced animations 
            // would require custom behaviors
        };
    }
}
