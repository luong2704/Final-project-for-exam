```csharp
using Campus.ViewModels;
using Campus.Services;

namespace Campus.Views;

public partial class CategoryFilterView : ContentPage
{
    public CategoryFilterView()
    {
        InitializeComponent();
        
        // Get services from MauiProgram.cs service provider
        var categoryService = ServiceProvider.GetService<ICategoryService>();
        var eventService = ServiceProvider.GetService<IEventService>();
        
        if (categoryService != null && eventService != null)
        {
            BindingContext = new CategoryViewModel(categoryService, eventService);
        }
    }

    protected override async void OnAppearing()
    {
        base.OnAppearing();
        
        if (BindingContext is CategoryViewModel viewModel)
        {
            await viewModel.LoadCategoriesAsync();
            await viewModel.LoadEventsAsync();
        }
    }
}
```
