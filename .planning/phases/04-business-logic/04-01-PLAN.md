/// <summary>
/// Filters events by exact category name match (case-insensitive)
/// </summary>
public static List<Event> FilterEventsByCategory(IEnumerable<Event> events, string categoryName)
{
    if (events == null || string.IsNullOrWhiteSpace(categoryName))
    {
        return events?.ToList() ?? new List<Event>();
    }
    
    return events
        .Where(e => e.Category?.Equals(categoryName, StringComparison.OrdinalIgnoreCase) == true)
        .ToList();
}

/// <summary>
/// Filters events by CategoryType - matches events whose Category string
/// matches any Category with that type
/// </summary>
public static List<Event> FilterEventsByType(IEnumerable<Event> events, CategoryType type)
{
    if (events == null)
    {
        return new List<Event>();
    }
    
    // Get category names that belong to the specified type
    var categoryNamesOfType = _categories
        .Where(c => c.CategoryType == type)
        .Select(c => c.CategoryName.ToLowerInvariant())
        .ToHashSet();
    
    return events
        .Where(e => !string.IsNullOrEmpty(e.Category) && 
                    categoryNamesOfType.Contains(e.Category.ToLowerInvariant()))
        .ToList();
}
```

[RelayCommand]
private void ApplyFilter()
{
    if (SelectedCategory != null)
    {
        IsFiltered = true;
        if (Events.Any())
        {
            var filtered = CategoryService.FilterEventsByType(Events, SelectedCategory.CategoryType);
            FilteredEvents.Clear();
            foreach (var e in filtered)
            {
                FilteredEvents.Add(e);
            }
        }
    }
}
```

5. Update ClearFilter method:
```csharp
[RelayCommand]
private void ClearFilter()
{
    SelectedCategory = null;
    IsFiltered = false;
    
    // Restore original event list
    FilteredEvents.Clear();
    foreach (var e in Events)
    {
        FilteredEvents.Add(e);
    }
}
```

public async Task LoadEventsAsync()

    Events.Clear();
    foreach (var e in events)
    {
        Events.Add(e);
    }
    
    // Initialize filtered events with all events

        FilteredEvents.Add(e);
